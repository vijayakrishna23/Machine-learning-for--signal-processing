def hann_window(raw_data, m_size):
    hann_w = np.hanning(m_size)
    new_data = []
    
    counter = 0
    
    while counter + m_size <= raw_data.shape[0]:
        data_window = np.multiply(raw_data[counter:counter+m_size], hann_w)
        new_data.append(data_window)
        
        counter += int(m_size/2)
    
    new_data_matrix = np.array(new_data).T
    
    return new_data_matrix

def reversing_overlap(X, m_size):

    X_out = X[:(X.shape[0] - int(m_size/2)),0]
    for col in range(1,X.shape[1]):
        X_col = X[(X.shape[0] - int(m_size/2)):,col-1] + X[0:int(m_size/2),col]
        X_out = np.concatenate((X_out,X_col))

    X_out = X_out.reshape(-1) 

    return X_out

def k_Means_Clustering(data, number_clusters, max_iter, tolerance):
    
    #randomly initiating two cluster centroids
    clusters = np.random.uniform(low = int(min(data)), high = int(max(data)), size = number_clusters).reshape(-1,1)  
    cluster_matrix = np.array([[0.0] * number_clusters] * data.shape[0])

    iter_count = 0    
    while iter_count <= max_iter:

        for i in range(0, len(data)):
            if (np.abs(data[i] - clusters[0])) < (np.abs(data[i] - clusters[1])):
                cluster_matrix[i][0] = 1
                cluster_matrix[i][1] = 0
            else:
                cluster_matrix[i][1] = 1
                cluster_matrix[i][0] = 0
        
        cluster_previous = clusters
        
        cluster_centroids_dot = np.dot(data.T, cluster_matrix)
        total_sum = np.sum(cluster_matrix, axis = 0)

        clusters = (cluster_centroids_dot/total_sum).reshape(-1)
        print(clusters)
        
        iter_count += 1
        
        #Early stopping
        if np.abs(clusters[0] - cluster_previous[0]) < tolerance and np.abs(clusters[1] - cluster_previous[1]) < tolerance:
            break
    
    #Below method to create dataframe for visualisation purpose referenced from 
    # https://tomaugspurger.github.io/modern-6-visualization.html
    #Start
    result_df = pd.DataFrame(data)
    
    result_df.columns = ['Disparity Data']
    result_df['Cluster'] = np.argmax(cluster_matrix, axis = 1)
    #End 
    
    return clusters, result_df

def pdf_dist_normal(mu, sigma, x):
    pdf = (1/np.sqrt((2*np.pi*(sigma**2)))) * (np.exp(-(((x-mu)**2)/(2*(sigma**2)))))
    return pdf

def EM(data, num_clusters, max_iter, tolerance):
    #Initialised probability of the number of stars in clusters by visually observing
    prob_A, prob_B = 0.1, 0.9
    
    #Using the centroids found using k-means clustering
    mean_A, mean_B = 27.743, 42.312
    sd_A, sd_B = 2, 2
    
    iter_count = 0
    
    while iter_count <= max_iter:
        probX_given_A = pdf_dist_normal(mean_A, sd_A, data)
        probX_given_B = pdf_dist_normal(mean_B, sd_B, data)
        probX_given_A[probX_given_A == 0] = 0.05
        probX_given_B[probX_given_B == 0] = 0.05
        
        #E-step
        likelihood_A = (prob_A * probX_given_A)/((prob_A * probX_given_A) + (prob_B * probX_given_B))
        likelihood_B = (prob_B * probX_given_B)/((prob_A * probX_given_A) + (prob_B * probX_given_B))
        
        prev_mean_A = mean_A
        prev_mean_B = mean_B
        
        #M-step
        mean_A = np.sum(likelihood_A * data)/np.sum(likelihood_A)
        mean_B = np.sum(likelihood_B * data)/np.sum(likelihood_B)
        
        sd_A = np.sqrt(np.sum(likelihood_A * (data - mean_A) * (data - mean_A))/np.sum(likelihood_A))
        sd_B = np.sqrt(np.sum(likelihood_B * (data - mean_B) * (data - mean_B))/np.sum(likelihood_B))
        
        iter_count += 1
        
        print(mean_A, mean_B)
        
        if np.abs(mean_A - prev_mean_A) < tolerance and np.abs(mean_B - prev_mean_B) < tolerance:
            break
            
    cluster_matrix = np.argmax(np.concatenate((likelihood_A, likelihood_B), axis = 1), axis = 1)
    EM_df = pd.DataFrame(data)
    EM_df.columns = ['Disparity Data']
    EM_df['Cluster'] = cluster_matrix
    
    return mean_A, mean_B, EM_df
